50      @Override
51      public void init() throws Exception {
52          Properties properties = new Properties();
53  
54          properties.load(new FileInputStream("config.properties"));
55  
56          String userFile = properties.getProperty("userFile");
57          String itemFile = properties.getProperty("itemFile");
58  
59          FileUserDao userDao = new FileUserDao(userFile);
60          FileItemDao itemDao = new FileItemDao(itemFile, userDao);
61          itemList = new ItemList(itemDao, userDao);
62      }

64      public Node createItemNode(Item item) {
65          HBox box = new HBox(10);
66          Label label = new Label(item.getProduct());
67          label.setMinHeight(28);
68          Button button = new Button("x");
69          button.setOnAction(e -> {
70              itemList.packed(item.getId());
71              redrawItemList();
72          });
73  
74          Region spacer = new Region();
75          HBox.setHgrow(spacer, Priority.ALWAYS);
76          box.setPadding(new Insets(0, 5, 0, 5));
77  
78          box.getChildren().addAll(label, spacer, button);
79          return box;
80      }
81  
82      public void redrawItemList() {
83          itemNode.getChildren().clear();
84  
85          List<Item> unPackedItems = itemList.unPacked();
86          unPackedItems.forEach(item -> {
87              itemNode.getChildren().add(createItemNode(item));
88          });
89      }

91      public void login(Stage stage) {
92          //login Scene
93          VBox loginPane = new VBox(10);
94          HBox inputPane = new HBox(10);
95          loginPane.setPadding(new Insets(10));
96          Label loginLabel = new Label("Tunnus:");
97          TextField usernameInput = new TextField();
98  
99          inputPane.getChildren().addAll(loginLabel, usernameInput);
100 
101         Button loginButton = new Button("Kirjaudu");
102         Button createButton = new Button("Luo uusi käyttäjä");
103         loginButton.setOnAction(e -> {
104             String username = usernameInput.getText();
105             menuLabel.setText("Tervetuloa " + username + " !");
106             if (itemList.login(username)) {
107                 loginMessage.setText("Tervetuloa");
108                 stage.setScene(itemListScene);
109                 usernameInput.setText("");
110                 redrawItemList();
111             } else {
112                 loginMessage.setText("Käyttäjää ei löydy");
113                 loginMessage.setTextFill(Color.BLUEVIOLET);
114             }
115 
116         });
117         createButton.setOnAction(e -> {
118             usernameInput.setText("");
119             stage.setScene(newUserScene);
120         });
121 
122         loginPane.getChildren().addAll(loginMessage, inputPane, loginButton, createButton);
123 
124         loginScene = new Scene(loginPane, 400, 300);
125     }

127     public void createNewUserScene(Stage stage) {
128         //create new user
129         VBox newUserPane = new VBox(10);
130 
131         HBox newUsernamePane = new HBox(10);
132         newUsernamePane.setPadding(new Insets(10));
133         TextField newUsernameInput = new TextField();
134         Label newUsernameLabel = new Label("Tunnus:");
135         newUsernameLabel.setPrefWidth(100);
136         newUsernamePane.getChildren().addAll(newUsernameLabel, newUsernameInput);
137 
138         HBox newNamePane = new HBox(10);
139         newNamePane.setPadding(new Insets(10));
140         TextField newNameInput = new TextField();
141         Label newNameLabel = new Label("Nimi:");
142         newNameLabel.setPrefWidth(100);
143         newNamePane.getChildren().addAll(newNameLabel, newNameInput);
144 
145         Label userCreationMessage = new Label();
146 
147         Button createNewUserButton = new Button("Luo");
148         createNewUserButton.setPadding(new Insets(10));
149 
150         createNewUserButton.setOnAction(e -> {
151             String username = newUsernameInput.getText();
152             String name = newNameInput.getText();
153 
154             if (username.length() == 2 || name.length() < 2) {
155                 userCreationMessage.setText("Käyttäjätunnus tai nimen pituus on liian lyhyt");
156                 userCreationMessage.setTextFill(Color.BLUEVIOLET);
157             } else if (itemList.createUser(username, name)) {
158                 userCreationMessage.setText("");
159                 loginMessage.setText("Uusi käyttäjä luotu onnistuneesti, nyt voit kirjautua sisään");
160                 loginMessage.setTextFill(Color.GREEN);
161                 stage.setScene(loginScene);
162             } else {
163                 userCreationMessage.setText("Tunnus on jo käytössä kokeile toista.");
164                 userCreationMessage.setTextFill(Color.BLUEVIOLET);
165             }
166 
167         });
168 
169         newUserPane.getChildren().addAll(userCreationMessage, newUsernamePane, newNamePane, createNewUserButton);
170 
171         newUserScene = new Scene(newUserPane, 400, 300);
172     }

174     public void mainScene(Stage stage) {
175         //main
176         ScrollPane itemScrollBar = new ScrollPane();
177         BorderPane mainPane = new BorderPane(itemScrollBar);
178         itemListScene = new Scene(mainPane, 400, 500, Color.CORAL);
179 
180         HBox menuPane = new HBox(10);
181         Region menuSpacer = new Region();
182         HBox.setHgrow(menuSpacer, Priority.ALWAYS);
183         Button logoutButton = new Button("Kirjaudu ulos");
184         menuPane.getChildren().addAll(menuLabel, menuSpacer, logoutButton);
185         logoutButton.setOnAction(e -> {
186             itemList.logout();
187             stage.setScene(loginScene);
188         });
189 
190         HBox createForm = new HBox(10);
191         Button createItem = new Button("Lisää");
192 
193         Region spacer = new Region();
194         HBox.setHgrow(spacer, Priority.ALWAYS);
195         TextField newItemInput = new TextField();
196 
197         createForm.getChildren().addAll(newItemInput, spacer, createItem);
198 
199         itemNode = new VBox(10);
200         itemNode.setMaxWidth(280);
201         itemNode.setMinWidth(280);
202         redrawItemList();
203 
204         itemScrollBar.setContent(itemNode);
205         mainPane.setBottom(createForm);
206         mainPane.setTop(menuPane);
207 
208         createItem.setOnAction(e -> {
209             itemList.createItem(newItemInput.getText());
210             newItemInput.setText("");
211             redrawItemList();
212         });
213     }
